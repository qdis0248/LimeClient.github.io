import net.minecraft.client.Minecraft;
import com.Lime.module.*;
import com.Lime.anticheat.BypassEngine;

public class LimeCore {
    private final Minecraft mc = Minecraft.getMinecraft();
    private final ModuleManager moduleManager = new ModuleManager();
    private final BypassEngine bypassEngine = new BypassEngine();

    public void initClient() {
        moduleManager.registerModule(new ESPModule("Player ESP", true, Color.WHITE));
        moduleManager.registerModule(new AimBotModule("Aim Bot", false, 0.15f));
        moduleManager.registerModule(new FlyModule("Creative Fly", false, 0.6f));
        moduleManager.registerModule(new FastBreakModule("Fast Break", true));
        moduleManager.registerModule(new NoFallModule("No Fall Damage", true));

        bypassEngine.enableBypass("Hypixel", "Mineplex", "CubeCraft");
        bypassEngine.setSpoofDelay(50);

        loadConfig("Lime_config.json");
        
        System.out.println("[Lime] 客户端初始化完成 - 安全等级: HIGH");
    }

    private void loadConfig(String path) {
        if (mc.fileManager.exists(path)) {
            Config config = new Config(path);
            moduleManager.applyConfig(config);
            bypassEngine.applyConfig(config);
        }
    }

    public boolean isModuleEnabled(String moduleName) {
        return moduleManager.getModuleByName(moduleName).isEnabled();
    }
}

public abstract class Module {
    private final String name;
    private boolean enabled;

    public Module(String name, boolean enabled) {
        this.name = name;
        this.enabled = enabled;
    }

    public abstract void onUpdate();
    public abstract void onRender();

    public void toggle() {
        this.enabled = !this.enabled;
        if (enabled) onEnable();
        else onDisable();
    }

    protected void onEnable() {}
    protected void onDisable() {}
}
